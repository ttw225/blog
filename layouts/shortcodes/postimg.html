{{/* Resolve assets/img// for bilingual posts (slug from front matter or file base name). Usage: {{< postimg "screenshot.png" >}} {{< postimg src="a.png" alt="..." caption="..." >}} Mirrors Blowfish figure shortcode behaviour for local assets. */}} {{- $fname := "" -}} {{- if .IsNamedParams -}} {{- $fname = .Get "src" -}} {{- else -}} {{- $fname = .Get 0 -}} {{- end -}} {{- if not $fname -}} {{- errorf "postimg: pass a filename, e.g. {{< postimg \"shot.png\" >}} (page %q)" .Page.File.Path -}} {{- end -}} {{- $slug := .Page.Params.slug | default .Page.File.ContentBaseName -}} {{- $absPath := printf "img/%s/%s" $slug $fname -}} {{- $alt := .Get "alt" | default $fname -}} {{- $caption := .Get "caption" -}} {{- $href := .Get "href" -}} {{- $class := .Get "class" -}} {{- $figureClass := .Get "figureClass" -}} {{- $target := .Get "target" | default "_blank" -}} {{- $nozoom := .Get "nozoom" | default false -}} {{- $disableImageOptimization := .Site.Params.disableImageOptimization | default false -}} {{- $url := urls.Parse $absPath -}} {{- with $href }}{{ end -}} {{- if findRE "^https?" $url.Scheme }} {{- else }} {{- $resource := "" }} {{- if $.Page.Resources.GetMatch $absPath }} {{- $resource = $.Page.Resources.GetMatch $absPath }} {{- else if resources.Get $absPath }} {{- $resource = resources.Get $absPath }} {{- end }} {{- with $resource }} {{- if or $disableImageOptimization (eq .MediaType.SubType "svg") }} {{- else }} {{- end }} {{- else }} {{- end }} {{- end }} {{ if $href }}{{ end }} {{ with $caption }}{{ end }}
